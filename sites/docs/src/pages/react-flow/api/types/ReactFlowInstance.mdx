# ReactFlowInstance&lt;NodeData, EdgeData>

```ts
export type ReactFlowInstance<NodeData = any, EdgeData = any> = {
  // Nodes and Edges
  getNode: (id: string) => Node<NodeData> | undefined;
  getNodes: () => Node<NodeData>[];
  addNodes: (payload: Node<NodeData>[] | Node<NodeData>) => void;
  setNodes: (
    payload: Node<NodeData>[] | ((nodes: Node<NodeData>[]) => Node<NodeData>[])
  ) => void;

  getEdge: (id: string) => Edge<EdgeData> | undefined;
  getEdges: () => Edge<EdgeData>[];
  addEdges: (payload: Edge<EdgeData>[] | Edge<EdgeData>) => void;
  setEdges: (
    payload: Edge<EdgeData>[] | ((edges: Edge<EdgeData>[]) => Edge<EdgeData>[])
  ) => void;

  toObject: () => ReactFlowJsonObject<NodeData, EdgeData>;
  deleteElements: ({ nodes, edges }: DeleteElementsOptions) => void;

  // Intersections
  getIntersectingNodes: (
    node: (Partial<Node<NodeData>> & { id: Node['id'] }) | Rect,
    partially?: boolean,
    nodes?: Node<NodeData>[]
  ) => Node<NodeData>[];

  isNodeIntersecting: (
    node: (Partial<Node<NodeData>> & { id: Node['id'] }) | Rect,
    area: Rect,
    partially?: boolean
  ) => boolean;

  // Viewport
  viewportInitialized: boolean;
  zoomIn: ZoomInOut;
  zoomOut: ZoomInOut;
  zoomTo: ZoomTo;
  getZoom: GetZoom;
  setViewport: SetViewport;
  getViewport: GetViewport;
  fitView: FitView;
  setCenter: SetCenter;
  fitBounds: FitBounds;
  project: Project;
};
```
