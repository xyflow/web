## Local Installation

You need to have [Node.js](https://nodejs.org/en/) installed and a package manager of your choice:
[npm](https://www.npmjs.com/), [pnpm](https://pnpm.io/), [yarn](https://yarnpkg.com/), [bun](https://bun.sh/)

### Vite template

If you want a preconfigured codebase, you can use our [Vite template](https://github.com/xyflow/vite-react-flow-template):

```bash copy npm2yarn
npx degit xyflow/vite-react-flow-template app-name
```

### Manual setup

First, spin up a new [React](https://reactjs.org/) project however you like; we
recommend using [Vite](https://vitejs.dev/) but the choice is yours.

```bash copy npm2yarn
npm init vite my-react-flow-app -- --template react
```

React Flow is published on npm as [`@xyflow/react`](https://npmjs.com/package/@xyflow/react). Change directory into your project folder and add it next.

```bash copy npm2yarn
npm install @xyflow/react
```

Lastly, spin up the dev server and you're good to go!

## Creating your first flow

First we import the [`<ReactFlow />`](/api-reference/react-flow#reactflow) component from the `@xyflow/react` package.
That and a handful of node and edge objects are all we need to get something going! Get rid of everything inside `App.jsx` and add the following:

```jsx
import { ReactFlow } from '@xyflow/react';
import '@xyflow/react/dist/style.css'; // required

const initialNodes = [
  { id: 'n1', position: { x: 0, y: 0 }, data: { label: 'Node 1' } },
  { id: 'n2', position: { x: 0, y: 100 }, data: { label: 'Node 2' } },
];
const initialEdges = [{ id: 'n1-n2', source: 'n1', target: 'n2' }];

export default function App() {
  return (
    <div style={{ width: '100vw', height: '100vh' }}>
      <ReactFlow nodes={initialNodes} edges={initialEdges} />
    </div>
  );
}
```

There are a few things to pay attention to here:

- <Emoji content="ðŸŽ¨" /> You must import the React Flow css stylesheet.
- <Emoji content="ðŸ“" /> The `<ReactFlow />` component must have a parent element with a
  width and height.

## Adding interactivity

Diagrams created with React Flow are fully interactive. We can move nodes around, connect, delete, and more. To get this basic functionality, we need to add three things:

- A callback for when [nodes change](/api-reference/react-flow#onnodeschange).
- A callback for when [edges change](/api-reference/react-flow#onedgeschange).
- A callback for when nodes are [connected](/api-reference/react-flow#onconnect).

```jsx
import { useState, useCallback } from 'react';
import { ReactFlow, applyNodeChanges, applyEdgeChanges, addEdge } from '@xyflow/react';
import '@xyflow/react/dist/style.css'; // required

const initialNodes = [
  { id: 'n1', position: { x: 0, y: 0 }, data: { label: 'Node 1' } },
  { id: 'n2', position: { x: 0, y: 100 }, data: { label: 'Node 2' } },
];
const initialEdges = [{ id: 'n1-n2', source: 'n1', target: 'n2' }];

export default function App() {
  const [nodes, setNodes] = useState(initialNodes);
  const [edges, setEdges] = useState(initialEdges);

  const onNodesChange = useCallback(
    (changes) => setNodes((nodesSnapshot) => applyNodeChanges(changes, nodesSnapshot)),
    [],
  );
  const onEdgesChange = useCallback(
    (changes) => setEdges((edgesSnapshot) => applyEdgeChanges(changes, edgesSnapshot)),
    [],
  );
  const onConnect = useCallback(
    (params) => setEdges((edgesSnapshot) => addEdge(params, edgesSnapshot)),
    [],
  );

  return (
    <div style={{ width: '100vw', height: '100vh' }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
      />
    </div>
  );
}
```

## Some extra goodies

Finally, React Flow ships with some plugins out of the box for things like a
[`<Minimap />`](/api-reference/components/minimap) and viewport
[`<Controls />`](/api-reference/components/controls).
Let's pass them as children to our `<ReactFlow />` component:

```jsx
import React, { useCallback } from 'react';
import {
  ReactFlow,
  MiniMap,
  Controls,
  Background,
  applyNodeChanges,
  applyEdgeChanges,
  addEdge,
} from '@xyflow/react';
import '@xyflow/react/dist/style.css';

const initialNodes = [
  { id: 'n1', position: { x: 0, y: 0 }, data: { label: 'Node 1' } },
  { id: 'n2', position: { x: 0, y: 100 }, data: { label: 'Node 2' } },
];
const initialEdges = [{ id: 'n1-n2', source: 'n1', target: 'n2' }];

export default function App() {
  const [nodes, setNodes] = useState(initialNodes);
  const [edges, setEdges] = useState(initialEdges);

  const onNodesChange = useCallback(
    (changes) => setNodes((nodesSnapshot) => applyNodeChanges(changes, nodesSnapshot)),
    [],
  );
  const onEdgesChange = useCallback(
    (changes) => setEdges((edgesSnapshot) => applyEdgeChanges(changes, edgesSnapshot)),
    [],
  );
  const onConnect = useCallback(
    (params) => setEdges((edgesSnapshot) => addEdge(params, edgesSnapshot)),
    [],
  );

  return (
    <div style={{ width: '100vw', height: '100vh' }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
      >
        <Controls />
        <MiniMap />
        <Background variant="dots" gap={12} size={1} />
      </ReactFlow>
    </div>
  );
}
```
